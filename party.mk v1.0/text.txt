    <script>
      function initMap() {
  // Set the coordinates for the map center
  const mapCenter = [41.9973, 21.4274];
  
  // Create the map
  const map = L.map("map").setView(mapCenter, 13);
  
  // Add the tile layer to the map
  L.tileLayer("https://{s}.tile.openstreetmap.org/{z}/{x}/{y}.png", {
    attribution: '&copy; <a href="https://www.openstreetmap.org/">OpenStreetMap</a> contributors',
  }).addTo(map);

  // Add a marker for the map center
  L.marker(mapCenter).addTo(map);
  
  // Make a request to the Geoapify Places API to get information about nightclubs in Skopje
  const url = `https://api.geoapify.com/v1/places?categories=nightlife:club&filter=countrycode:MK&limit=10&apiKey=${apiKey}&bbox=${map.getBounds().toBBoxString()}`;
  fetch(url)
    .then(response => response.json())
    .then(data => {
      // Add markers for each nightclub
      data.features.forEach(feature => {
        const marker = L.marker([feature.properties.lat, feature.properties.lon]).addTo(map);
        marker.bindPopup(feature.properties.name);
      });
    })
    .catch(error => {
      console.error('Error:', error);
    });
}

    </script>
